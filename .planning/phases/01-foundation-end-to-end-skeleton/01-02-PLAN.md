```typescript
export type CategoryKey =
  | 'businessProcess' | 'technology' | 'data' | 'aiAdoption' | 'automation'
  | 'aiAgents' | 'security' | 'people' | 'leadership' | 'innovation';

export type CategoryScores = Record<CategoryKey, number>;

export interface Stage {
  level: number;   // 1-6
  label: string;   // e.g. "Level 1 — AI Unaware"
}

export interface CompanyProfileAnswers {
  industry: string;
  employeeBand: string;
  revenueBand: string;
  businessModel: string;
  leadershipGoals: string[];
  blendedHourlyCost: number;
}

// The full shape the scoring engine reads. Sections not yet collected by
// Phase 1's 2-step wizard are optional and, in Phase 1, always undefined.
export interface ScoringInput {
  companyProfile: CompanyProfileAnswers;
  departments?: Record<string, { rating: number; capabilities: string[] }>;
  techStack?: { systems: string[]; apiMaturity: string };
  dataReadiness?: { quality: number; security: number; accessibility: number; knowledgeMgmt: number };
  aiAdoption?: { toolsInUse: string[]; frequency: string };
  automation?: { processesAutomated: string[] };
  agentInterest?: { agentTypes: string[] };
  security?: { checklist: string[] };
  employeeReadiness?: { aiSkills: number; training: number; changeReadiness: number; leadershipSupport: number; innovationCulture: number };
}
```

From `lib/constants/companyProfile.ts` (Plan 01-01): `LEADERSHIP_GOALS` — a
6-item `string[]` used as the denominator for the leadership-goals-selected
ratio.
